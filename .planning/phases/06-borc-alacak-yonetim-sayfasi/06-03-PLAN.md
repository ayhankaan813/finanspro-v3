  ```typescript
  function getHeatColor(amount: number, maxAmount: number): string {
    if (amount === 0 || maxAmount === 0) return '';
    const intensity = amount / maxAmount;
    if (intensity < 0.33) return 'bg-blue-50 text-blue-700';      // Dusuk
    if (intensity < 0.66) return 'bg-amber-50 text-amber-700';    // Orta
    return 'bg-red-50 text-red-700';                               // Yuksek
  }
  ```
